[TestFixture]
public class MemoryTests
{
    private MemoryMonitor memoryMonitor;
    
    [SetUp]
    public void Setup()
    {
        GameObject memoryObj = new GameObject();
        memoryMonitor = memoryObj.AddComponent<MemoryMonitor>();
    }
    
    [Test]
    public void TestMemoryLeak()
    {
        long initialMemory = memoryMonitor.GetTotalMemory();
        
        // 运行游戏10分钟
        for (int i = 0; i < 600; i++)
        {
            // 模拟游戏循环
            SimulateGameLoop();
            memoryMonitor.Update();
        }
        
        long finalMemory = memoryMonitor.GetTotalMemory();
        long memoryIncrease = finalMemory - initialMemory;
        
        // 内存增长不应超过100MB
        Assert.Less(memoryIncrease, 100 * 1024 * 1024, "Memory leak detected");
    }
    
    private void SimulateGameLoop()
    {
        // 模拟游戏循环
        ObjectPool.Instance.SpawnFromPool("Unit", Vector3.zero, Quaternion.identity);
        ObjectPool.Instance.SpawnFromPool("Unit", Vector3.one, Quaternion.identity);
    }
}

public class FeedbackAnalyzer : MonoBehaviour
{
    [System.Serializable]
    public class Feedback
    {
        public string playerId;
        public string feedbackText;
        public int rating;
        public FeedbackCategory category;
        public DateTime timestamp;
    }
    
    public enum FeedbackCategory
    {
        Bug,
        Balance,
        UI,
        Performance,
        Feature,
        Other
    }
    
    private List<Feedback> feedbacks = new List<Feedback>();
    
    public void AnalyzeFeedbacks()
    {
        // 按类别统计
        var categoryCounts = feedbacks
            .GroupBy(f => f.category)
            .Select(g => new { Category = g.Key, Count = g.Count() });
        
        // 计算平均评分
        float avgRating = feedbacks.Average(f => f.rating);
        
        // 分析高频问题
        var commonIssues = feedbacks
            .GroupBy(f => f.feedbackText)
            .Select(g => new { Issue = g.Key, Count = g.Count() })
            .OrderByDescending(g => g.Count)
            .Take(10);
        
        Debug.Log($"Average Rating: {avgRating:F2}");
        Debug.Log("Common Issues:");
        foreach (var issue in commonIssues)
        {
            Debug.Log($"{issue.Issue}: {issue.Count} reports");
        }
    }
}
