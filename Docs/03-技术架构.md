# 技术架构

## 1. 项目架构概览

### 1.1 技术栈选择

#### 1.1.1 游戏引擎
- **Unity 2021.3 LTS**
  - 稳定的长期支持版本
  - 优秀的3D渲染能力
  - 强大的物理引擎
  - 丰富的生态系统

#### 1.1.2 编程语言
- **C#**
  - Unity主要编程语言
  - 面向对象设计
  - 强类型系统
  - 优秀的性能

#### 1.1.3 网络框架
- **Mirror Networking** (推荐)
  - 高性能
  - 易于使用
  - 活跃的社区
  - 支持Unity最新版本

**备选方案**: Photon Unity Networking (PUN2)

#### 1.1.4 数据库
- **SQLite** (本地数据)
  - 轻量级

  - 适合本地存储

- **MySQL/PostgreSQL** (云端数据)
  - 玩家账户

#### 1.1.5 UI框架

  - Unity内置UI系统

- **TextMeshPro**

### 1.2 架构模式

#### 1.2.1 整体架构
采用**分层架构** + **组件化设计**：

```
┌─────────────────────────────────────┐
│         Presentation Layer          │  (UI, 场景管理)
├─────────────────────────────────────┤
│          Game Logic Layer           │  (游戏逻辑, AI, 战斗)
├─────────────────────────────────────┤
│           Network Layer             │  (网络同步, 匹配)
├─────────────────────────────────────┤
│           Data Layer               │  (数据存储, 配置)
```

#### 1.2.2 设计模式
- **单例模式**: 管理器类

- **工厂模式**: 卡牌和单位创建

```csharp
public class UnitAI : MonoBehaviour
{
    private Unit unit;
    private UnitStats stats;
    private List<Unit> enemiesInRange = new List<Unit>();
    private Unit currentTarget;
    
    public void Initialize(Unit unit)
    {
        this.unit = unit;
        this.stats = unit.stats;
    }
    
    void Update()
    {
        if (currentTarget == null || !IsTargetValid(currentTarget))
        {
            FindNewTarget();
        }
        
        if (currentTarget != null)
        {
            if (IsInRange(currentTarget))
            {
                AttackTarget();
            }
            else
            {
                MoveToTarget();
            }
        }
    }
    
    private void FindNewTarget()
    {
        // 寻找最近的敌人
        currentTarget = FindClosestEnemy();
    }
    
    private bool IsInRange(Unit target)
    {
        float distance = Vector3.Distance(unit.transform.position, target.transform.position);
        return distance <= stats.attackRange;
    }
}
```

```csharp
public class NetworkUnit : NetworkBehaviour
{
    [SyncVar] private Vector3 syncPosition;
    [SyncVar] private Quaternion syncRotation;
    [SyncVar] private int syncHealth;
    
    private float lerpRate = 15f;
    
    void Update()
    {
        if (isServer)
        {
            // 服务器更新位置
            syncPosition = transform.position;
            syncRotation = transform.rotation;
        }
        else
        {
            // 客户端插值同步
            SyncTransform();
        }
    }
    
    private void SyncTransform()
    {
        transform.position = Vector3.Lerp(transform.position, syncPosition, Time.deltaTime * lerpRate);
        transform.rotation = Quaternion.Slerp(transform.rotation, syncRotation, Time.deltaTime * lerpRate);
    }
    
    [Command]
    public void CmdTakeDamage(int damage)
    {
        syncHealth -= damage;
        RpcTakeDamage(damage);
    }
    
    [ClientRpc]
```

```csharp
public class ElixirManager : NetworkBehaviour
{
    [SyncVar] public float elixir;
    public float elixirRegenRate = 1f;
    public float maxElixir = 10f;
    
    void Update()
    {
        if (isServer)
        {
            RegenerateElixir();
        }
    }
    
    private void RegenerateElixir()
    {
        if (elixir < maxElixir)
        {
            elixir += elixirRegenRate * Time.deltaTime;
            elixir = Mathf.Min(elixir, maxElixir);
        }
    }
    
    public bool CanAfford(int cost)
    {
        return elixir >= cost;
    }
    
    public void SpendElixir(int cost)
    {
        if (CanAfford(cost))
        {
            elixir -= cost;
        }
    }
}
```

### 4.3 匹配系统

#### 4.3.1 匹配算法
```csharp
public class MatchMaker : MonoBehaviour
{
    public class MatchRequest
    {
        public NetworkConnection connection;
        public int trophies;
        public DateTime requestTime;
    }
    
    private Queue<MatchRequest> matchQueue = new Queue<MatchRequest>();
    private int maxTrophyDifference = 200;
    private int maxWaitTime = 30;
    
    public void RequestMatch(NetworkConnection connection, int trophies)
    {
        MatchRequest request = new MatchRequest
        {
            connection = connection,
            trophies = trophies,
            requestTime = DateTime.Now
        };
        
        matchQueue.Enqueue(request);
        TryMatch();
    }
    
    private void TryMatch()
    {
        if (matchQueue.Count < 2)
            return;
        
        MatchRequest player1 = matchQueue.Peek();
        
        foreach (MatchRequest player2 in matchQueue)
        {
            if (player1 == player2)
                continue;
            
            if (IsMatchValid(player1, player2))
            {
                CreateMatch(player1, player2);
                return;
            }
        }
        
        // 检查超时
        if ((DateTime.Now - player1.requestTime).TotalSeconds > maxWaitTime)
        {
            // 超时后放宽匹配条件
            CreateMatch(player1, matchQueue.ElementAt(1));
        }
    }
    
    private bool IsMatchValid(MatchRequest p1, MatchRequest p2)
    {
        int trophyDiff = Mathf.Abs(p1.trophies - p2.trophies);
        return trophyDiff <= maxTrophyDifference;
    }
    
    private void CreateMatch(MatchRequest p1, MatchRequest p2)
    {
        matchQueue.Dequeue();
        matchQueue.Dequeue();
        
        // 创建房间
        RoomManager.Instance.CreateRoom(p1.connection, p2.connection);
    }
}
```

```csharp
public class DatabaseManager : MonoBehaviour
{

    {

        connection.Open();
    }
    
    private void CreateTables()
    {
        string createCardsTable = @"
            CREATE TABLE IF NOT EXISTS cards (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                card_name TEXT NOT NULL,
                card_type INTEGER NOT NULL,
                elixir_cost INTEGER NOT NULL,
                level INTEGER NOT NULL,
                fragments INTEGER NOT NULL
            )";
        
        connection.ExecuteNonQuery(createCardsTable);
        
        // 创建其他表...
    }
    
    public List<CardData> LoadPlayerCards()
    {
        string query = "SELECT * FROM cards";
        SQLiteDataReader reader = connection.ExecuteQuery(query);
        
        List<CardData> cards = new List<CardData>();
        while (reader.Read())
        {
            CardData card = ParseCardFromReader(reader);
            cards.Add(card);
        }
        
        return cards;
    }
    
    public void SaveCard(CardData card)
    {
        string query = @"
            INSERT OR REPLACE INTO cards 
            (card_name, card_type, elixir_cost, level, fragments)
            VALUES (?, ?, ?, ?, ?)";
        
        connection.ExecuteNonQuery(query, 
            card.cardName, 
            (int)card.cardType, 
            card.elixirCost, 
            card.level, 
            card.fragments);
    }
}
```

### 6.1 对象池系统

#### 6.1.1 单位对象池
```csharp
public class UnitPool : MonoBehaviour
{
    [System.Serializable]
    public class Pool
    {
        public string tag;
        public GameObject prefab;
        public int size;
    }
    
    public List<Pool> pools;
    private Dictionary<string, Queue<GameObject>> poolDictionary;
    
    void Start()
    {
        poolDictionary = new Dictionary<string, Queue<GameObject>>();
        
        foreach (Pool pool in pools)
        {
            Queue<GameObject> objectPool = new Queue<GameObject>();
            
            for (int i = 0; i < pool.size; i++)
            {
                GameObject obj = Instantiate(pool.prefab);
                obj.SetActive(false);
                objectPool.Enqueue(obj);
            }
            
            poolDictionary.Add(pool.tag, objectPool);
        }
    }
    
    public GameObject SpawnFromPool(string tag, Vector3 position, Quaternion rotation)
    {
        if (!poolDictionary.ContainsKey(tag))
        {
            Debug.LogWarning($"Pool with tag {tag} doesn't exist");
            return null;
        }
        
        GameObject objectToSpawn = poolDictionary[tag].Dequeue();
        
        objectToSpawn.SetActive(true);
        objectToSpawn.transform.position = position;
        objectToSpawn.transform.rotation = rotation;
        
        poolDictionary[tag].Enqueue(objectToSpawn);
        
        return objectToSpawn;
    }
}
```

```csharp
public class ResourceManager : MonoBehaviour
{
    public static ResourceManager Instance { get; private set; }
    
    private Dictionary<string, Object> loadedResources = new Dictionary<string, Object>();
    
    void Awake()
    {
        Instance = this;
    }
    
    public void LoadResourceAsync<T>(string path, Action<T> callback) where T : Object
    {
        if (loadedResources.ContainsKey(path))
        {
            callback?.Invoke(loadedResources[path] as T);
            return;
        }
        
        StartCoroutine(LoadResourceCoroutine(path, callback));
    }
    
    private IEnumerator LoadResourceCoroutine<T>(string path, Action<T> callback) where T : Object
    {
        ResourceRequest request = Resources.LoadAsync<T>(path);
        
        yield return request;
        
        if (request.asset != null)
        {
            loadedResources[path] = request.asset;
            callback?.Invoke(request.asset as T);
        }
        else
        {
            Debug.LogError($"Failed to load resource at path: {path}");
        }
    }
    
    public void UnloadResource(string path)
    {
        if (loadedResources.ContainsKey(path))
        {
            Resources.UnloadAsset(loadedResources[path]);
            loadedResources.Remove(path);
        }
    }
}
```
