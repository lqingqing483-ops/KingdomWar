```csharp
public class DeckManager : MonoBehaviour
{
    public const int DECK_SIZE = 8;
    
    [SerializeField] private List<CardData> currentDeck = new List<CardData>();
    private List<CardData> ownedCards = new List<CardData>();
    
    public event Action<List<CardData>> OnDeckChanged;
    
    public bool AddCardToDeck(CardData card)
    {
        if (currentDeck.Count >= DECK_SIZE)
        {
            Debug.LogWarning("卡组已满");
            return false;
        }
        
        if (currentDeck.Contains(card))
        {
            Debug.LogWarning("卡组中已有该卡牌");
            return false;
        }
        
        currentDeck.Add(card);
        OnDeckChanged?.Invoke(currentDeck);
        return true;
    }
    
    public bool RemoveCardFromDeck(CardData card)
    {
        if (currentDeck.Remove(card))
        {
            OnDeckChanged?.Invoke(currentDeck);
            return true;
        }
        return false;
    }
    
    public bool SwapCards(CardData cardToRemove, CardData cardToAdd)
    {
        if (!currentDeck.Contains(cardToRemove))
            return false;
        
        if (currentDeck.Contains(cardToAdd))
            return false;
        
        int index = currentDeck.IndexOf(cardToRemove);
        currentDeck[index] = cardToAdd;
        OnDeckChanged?.Invoke(currentDeck);
        return true;
    }
    
    public List<CardData> GetCurrentDeck()
    {
        return new List<CardData>(currentDeck);
    }
    
    public float GetAverageElixirCost()
    {
        if (currentDeck.Count == 0)
            return 0f;
        
        float total = 0f;
        foreach (var card in currentDeck)
        {
            total += card.elixirCost;
        }
        return total / currentDeck.Count;
    }
    
    public void SaveDeck()
    {
        string deckJson = JsonUtility.ToJson(new DeckData(currentDeck));
        PlayerPrefs.SetString("CurrentDeck", deckJson);
        PlayerPrefs.Save();
    }
    
    public void LoadDeck()
    {
        string deckJson = PlayerPrefs.GetString("CurrentDeck", "");
        if (!string.IsNullOrEmpty(deckJson))
        {
            DeckData deckData = JsonUtility.FromJson<DeckData>(deckJson);
            currentDeck = deckData.cardNames.Select(name => 
                CardDatabase.Instance.GetCardByName(name)).ToList();
        }
    }
    
    [System.Serializable]
    private class DeckData
    {
        public List<string> cardNames;
        
        public DeckData(List<CardData> cards)
        {
            cardNames = cards.Select(c => c.cardName).ToList();
        }
    }
}
```

#### 1.2.2 CardCollection
```csharp
public class CardCollection : MonoBehaviour
{
    [System.Serializable]
    public class CardProgress
    {
        public string cardName;
        public int level;
        public int fragments;
        public int count;
    }
    
    private List<CardProgress> collection = new List<CardProgress>();
    
    public event Action<CardData> OnCardUnlocked;
    public event Action<CardData, int> OnCardUpgraded;
    
    public void AddCard(CardData card, int amount = 1)
    {
        CardProgress progress = collection.FirstOrDefault(c => c.cardName == card.cardName);
        
        if (progress == null)
        {
            progress = new CardProgress
            {
                cardName = card.cardName,
                level = 1,
                fragments = 0,
                count = amount
            };
            collection.Add(progress);
            OnCardUnlocked?.Invoke(card);
        }
        else
        {
            progress.count += amount;
        }
        
        SaveCollection();
    }
    
    public bool UpgradeCard(CardData card)
    {
        CardProgress progress = collection.FirstOrDefault(c => c.cardName == card.cardName);
        
        if (progress == null || progress.level >= card.maxLevel)
            return false;
        
        UpgradeCurve upgrade = card.upgradeCurves.FirstOrDefault(u => u.level == progress.level + 1);
        
        if (upgrade == null)
            return false;
        
        if (progress.fragments < upgrade.fragmentsRequired)
            return false;
        
        progress.fragments -= upgrade.fragmentsRequired;
        progress.level++;
        
        OnCardUpgraded?.Invoke(card, progress.level);
        SaveCollection();
        return true;
    }
    
    public int GetCardLevel(CardData card)
    {
        CardProgress progress = collection.FirstOrDefault(c => c.cardName == card.cardName);
        return progress?.level ?? 0;
    }
    
    public int GetCardCount(CardData card)
    {
        CardProgress progress = collection.FirstOrDefault(c => c.cardName == card.cardName);
        return progress?.count ?? 0;
    }
    
    public List<CardData> GetUnlockedCards()
    {
        return collection
            .Where(p => p.count > 0)
            .Select(p => CardDatabase.Instance.GetCardByName(p.cardName))
            .ToList();
    }
    
    private void SaveCollection()
    {
        string json = JsonUtility.ToJson(new CollectionData(collection));
        PlayerPrefs.SetString("CardCollection", json);
        PlayerPrefs.Save();
    }
    
    private void LoadCollection()
    {
        string json = PlayerPrefs.GetString("CardCollection", "");
        if (!string.IsNullOrEmpty(json))
        {
            CollectionData data = JsonUtility.FromJson<CollectionData>(json);
            collection = data.cards;
        }
    }
    
    [System.Serializable]
    private class CollectionData
    {
        public List<CardProgress> cards;
        
        public CollectionData(List<CardProgress> cards)
        {
            this.cards = cards;
        }
    }
}
```

### 1.3 卡牌部署系统

#### 1.3.1 CardSpawner
```csharp
public class CardSpawner : MonoBehaviour
{
    [SerializeField] private LayerMask spawnLayer;
    [SerializeField] private GameObject invalidSpawnIndicator;
    
    private CardData selectedCard;
    private GameObject previewObject;
    private bool isValidSpawn;
    
    public void SelectCard(CardData card)
    {
        selectedCard = card;
        CreatePreview();
    }
    
    public void DeselectCard()
    {
        selectedCard = null;
        DestroyPreview();
    }
    
    private void CreatePreview()
    {
        if (selectedCard == null)
            return;
        
        if (selectedCard.cardType == CardType.Unit)
        {
            previewObject = Instantiate(selectedCard.cardPrefab);
            previewObject.GetComponent<Unit>().enabled = false;
            SetPreviewTransparent();
        }
        else if (selectedCard.cardType == CardType.Building)
        {
            previewObject = Instantiate(selectedCard.buildingData.buildingPrefab);
            SetPreviewTransparent();
        }
    }
    
    private void SetPreviewTransparent()
    {
        Renderer[] renderers = previewObject.GetComponentsInChildren<Renderer>();
        foreach (var renderer in renderers)
        {
            foreach (Material mat in renderer.materials)
            {
                Color color = mat.color;
                color.a = 0.5f;
                mat.color = color;
            }
        }
    }
    
    private void DestroyPreview()
    {
        if (previewObject != null)
        {
            Destroy(previewObject);
            previewObject = null;
        }
    }
    
    void Update()
    {
        if (selectedCard == null || previewObject == null)
            return;
        
        UpdatePreviewPosition();
        CheckSpawnValidity();
    }
    
    private void UpdatePreviewPosition()
    {
        Ray ray = Camera.main.ScreenPointToRay(Input.mousePosition);
        RaycastHit hit;
        
        if (Physics.Raycast(ray, out hit, Mathf.Infinity, spawnLayer))
        {
            previewObject.transform.position = hit.point;
        }
    }
    
    private void CheckSpawnValidity()
    {
        isValidSpawn = CanSpawnAtPosition(previewObject.transform.position);
        
        if (invalidSpawnIndicator != null)
        {
            invalidSpawnIndicator.SetActive(!isValidSpawn);
        }
    }
    
    private bool CanSpawnAtPosition(Vector3 position)
    {
        if (selectedCard == null)
            return false;
        
        // 检查是否在己方区域
        if (!IsInOwnTerritory(position))
            return false;
        
        // 检查是否有足够空间
        if (IsPositionOccupied(position))
            return false;
        
        return true;
    }
    
    private bool IsInOwnTerritory(Vector3 position)
    {
        return position.z < 0;
    }
    
    private bool IsPositionOccupied(Vector3 position)
    {
        Collider[] colliders = Physics.OverlapSphere(position, 2f, spawnLayer);
        return colliders.Length > 0;
    }
    
    public bool TrySpawnCard()
    {
        if (selectedCard == null || !isValidSpawn)
            return false;
        
        if (selectedCard.cardType == CardType.Unit)
        {
            SpawnUnit(selectedCard, previewObject.transform.position);
        }
        else if (selectedCard.cardType == CardType.Building)
        {
            SpawnBuilding(selectedCard, previewObject.transform.position);
        }
        else if (selectedCard.cardType == CardType.Spell)
        {
            CastSpell(selectedCard, previewObject.transform.position);
        }
        
        DeselectCard();
        return true;
    }
    
    private void SpawnUnit(CardData card, Vector3 position)
    {
        GameObject unitObj = Instantiate(card.cardPrefab, position, Quaternion.identity);
        Unit unit = unitObj.GetComponent<Unit>();
        
        int level = CardCollection.Instance.GetCardLevel(card);
        UnitStats stats = CalculateUnitStats(card.unitData, level);
        
        unit.Initialize(stats);
        
        GameEvents.OnUnitSpawned?.Invoke(unit);
    }
    
    private void SpawnBuilding(CardData card, Vector3 position)
    {
        GameObject buildingObj = Instantiate(card.buildingData.buildingPrefab, position, Quaternion.identity);
        Building building = buildingObj.GetComponent<Building>();
        
        int level = CardCollection.Instance.GetCardLevel(card);
        BuildingStats stats = CalculateBuildingStats(card.buildingData, level);
        
        building.Initialize(stats);
        
        GameEvents.OnBuildingSpawned?.Invoke(building);
    }
    
    private void CastSpell(CardData card, Vector3 position)
    {
        Spell spell = Instantiate(card.spellData.spellPrefab, position, Quaternion.identity).GetComponent<Spell>();
        spell.Initialize(card.spellData);
        
        GameEvents.OnSpellCast?.Invoke(spell);
    }
    
    private UnitStats CalculateUnitStats(UnitData baseData, int level)
    {
        UpgradeCurve upgrade = CardDatabase.Instance.GetUpgradeCurve(baseData, level);
        
        UnitStats stats = new UnitStats
        {
            maxHealth = Mathf.RoundToInt(baseData.maxHealth * upgrade.healthMultiplier),
            attackDamage = Mathf.RoundToInt(baseData.attackDamage * upgrade.damageMultiplier),
            attackSpeed = baseData.attackSpeed,
            moveSpeed = baseData.moveSpeed,
            attackRange = baseData.attackRange,
            deployTime = baseData.deployTime,
            unitType = baseData.unitType,
            targetType = baseData.targetType,
            attackType = baseData.attackType,
            damageType = baseData.damageType,
            canFly = baseData.canFly,
            isSplashDamage = baseData.isSplashDamage,
            splashRadius = baseData.splashRadius
        };
        
        return stats;
    }
    
    private BuildingStats CalculateBuildingStats(BuildingData baseData, int level)
    {
        UpgradeCurve upgrade = CardDatabase.Instance.GetUpgradeCurve(baseData, level);
        
        BuildingStats stats = new BuildingStats
        {
            maxHealth = Mathf.RoundToInt(baseData.maxHealth * upgrade.healthMultiplier),
            attackDamage = Mathf.RoundToInt(baseData.attackDamage * upgrade.damageMultiplier),
            attackSpeed = baseData.attackSpeed,
            attackRange = baseData.attackRange,
            lifetime = baseData.lifetime,
            buildingType = baseData.buildingType,
            isDefensive = baseData.isDefensive,
            isResource = baseData.isResource,
            elixirGenerationRate = baseData.elixirGenerationRate,
            goldGenerationRate = baseData.goldGenerationRate
        };
        
        return stats;
    }
}
```

### 2.1 Unit基类

#### 2.1.1 Unit类
```csharp
public class Unit : MonoBehaviour
{
    [Header("组件")]
    [SerializeField] private UnitMovement movement;
    [SerializeField] private UnitCombat combat;
    [SerializeField] private UnitAI ai;
    [SerializeField] private UnitHealth health;
    [SerializeField] private UnitAnimation animation;
    
    private UnitStats stats;
    private int currentHealth;
    private bool isDead;
    private Unit owner;
    private List<Buff> activeBuffs = new List<Buff>();
    
    public UnitStats Stats => stats;
    public int CurrentHealth => currentHealth;
    public bool IsDead => isDead;
    public Unit Owner => owner;
    
    public event Action<Unit> OnDeath;
    public event Action<int, int> OnHealthChanged;
    
    public void Initialize(UnitStats stats, Unit owner = null)
    {
        this.stats = stats;
        this.owner = owner;
        currentHealth = stats.maxHealth;
        isDead = false;
        
        movement?.Initialize(stats.moveSpeed);
        combat?.Initialize(stats);
        ai?.Initialize(this);
        health?.Initialize(stats.maxHealth);
        animation?.Initialize();
        
        GameEvents.OnUnitSpawned?.Invoke(this);
    }
    
    public void TakeDamage(int damage, DamageType damageType = DamageType.Physical)
    {
        if (isDead)
            return;
        
        // 计算伤害减免
        int finalDamage = CalculateDamage(damage, damageType);
        
        currentHealth -= finalDamage;
        OnHealthChanged?.Invoke(currentHealth, stats.maxHealth);
        
        // 显示伤害数字
        DamageNumberManager.Instance.ShowDamageNumber(finalDamage, transform.position);
        
        if (currentHealth <= 0)
        {
            Die();
        }
    }
    
    private int CalculateDamage(int damage, DamageType damageType)
    {
        int finalDamage = damage;
        
        // 应用buff
        foreach (var buff in activeBuffs)
        {
            if (buff.type == BuffType.DamageReduction)
            {
                finalDamage = Mathf.RoundToInt(finalDamage * (1f - buff.value));
            }
        }
        
        // 检查伤害类型抗性
        if (damageType == DamageType.Magic && stats.damageType == DamageType.True)
        {
            // 真实伤害无视抗性
        }
        
        return Mathf.Max(0, finalDamage);
    }
    
    public void Heal(int amount)
    {
        if (isDead)
            return;
        
        currentHealth = Mathf.Min(currentHealth + amount, stats.maxHealth);
        OnHealthChanged?.Invoke(currentHealth, stats.maxHealth);
        
        // 显示治疗数字
        DamageNumberManager.Instance.ShowHealNumber(amount, transform.position);
    }
    
    public void AddBuff(Buff buff)
    {
        activeBuffs.Add(buff);
        ApplyBuff(buff);
    }
    
    public void RemoveBuff(Buff buff)
    {
        activeBuffs.Remove(buff);
        RemoveBuffEffect(buff);
    }
    
    private void ApplyBuff(Buff buff)
    {
        switch (buff.type)
        {
            case BuffType.SpeedBoost:
                movement.SetMoveSpeed(stats.moveSpeed * (1f + buff.value));
                break;
            case BuffType.DamageBoost:
                combat.SetDamageMultiplier(1f + buff.value);
                break;
            case BuffType.AttackSpeedBoost:
                combat.SetAttackSpeedMultiplier(1f + buff.value);
                break;
        }
    }
    
    private void RemoveBuffEffect(Buff buff)
    {
        switch (buff.type)
        {
            case BuffType.SpeedBoost:
                movement.SetMoveSpeed(stats.moveSpeed);
                break;
            case BuffType.DamageBoost:
                combat.SetDamageMultiplier(1f);
                break;
            case BuffType.AttackSpeedBoost:
                combat.SetAttackSpeedMultiplier(1f);
                break;
        }
    }
    
    private void Die()
    {
        if (isDead)
            return;
        
        isDead = true;
        
        // 触发死亡事件
        OnDeath?.Invoke(this);
        GameEvents.OnUnitDestroyed?.Invoke(this);
        
        // 播放死亡动画
        animation.PlayDeathAnimation();
        
        // 处理死亡效果
        HandleDeathEffect();
        
        // 延迟销毁
        Destroy(gameObject, 2f);
    }
    
    private void HandleDeathEffect()
    {
        switch (stats.deathEffect)
        {
            case DeathEffect.Split:
                SplitOnDeath();
                break;
            case DeathEffect.Explode:
                ExplodeOnDeath();
                break;
            case DeathEffect.Poison:
                PoisonOnDeath();
                break;
            case DeathEffect.Heal:
                HealOnDeath();
                break;
        }
    }
    
    private void SplitOnDeath()
    {
        for (int i = 0; i < stats.splitCount; i++)
        {
            Vector3 offset = new Vector3(Random.Range(-1f, 1f), 0, Random.Range(-1f, 1f));
            Vector3 spawnPos = transform.position + offset;
            
            CardData splitCard = CardDatabase.Instance.GetCardByName(stats.splitUnitName);
            if (splitCard != null)
            {
                UnitSpawner.Instance.SpawnUnit(splitCard, spawnPos);
            }
        }
    }
    
    private void ExplodeOnDeath()
    {
        Collider[] colliders = Physics.OverlapSphere(transform.position, stats.splashRadius);
        foreach (var collider in colliders)
        {
            Unit target = collider.GetComponent<Unit>();
            if (target != null && target != this)
            {
                target.TakeDamage(stats.attackDamage / 2);
            }
        }
        
        // 播放爆炸特效
        EffectManager.Instance.PlayExplosionEffect(transform.position);
    }
    
    private void PoisonOnDeath()
    {
        Collider[] colliders = Physics.OverlapSphere(transform.position, stats.splashRadius);
        foreach (var collider in colliders)
        {
            Unit target = collider.GetComponent<Unit>();
            if (target != null && target != this)
            {
                target.AddBuff(new Buff
                {
                    type = BuffType.Poison,
                    value = 0.1f,
                    duration = 5f
                });
            }
        }
    }
    
    private void HealOnDeath()
    {
        Collider[] colliders = Physics.OverlapSphere(transform.position, stats.splashRadius);
        foreach (var collider in colliders)
        {
            Unit target = collider.GetComponent<Unit>();
            if (target != null && target.Owner == this.Owner)
            {
                target.Heal(stats.attackDamage / 2);
            }
        }
    }
    
    void Update()
    {
        UpdateBuffs();
    }
    
    private void UpdateBuffs()
    {
        for (int i = activeBuffs.Count - 1; i >= 0; i--)
        {
            activeBuffs[i].duration -= Time.deltaTime;
            if (activeBuffs[i].duration <= 0)
            {
                RemoveBuff(activeBuffs[i]);
            }
        }
    }
}

    Poison,
    Freeze,
    Stun
```

```csharp
public class UnitMovement : MonoBehaviour
{
    private float moveSpeed;
    private float currentSpeed;
    private NavMeshAgent agent;
    private Unit unit;
    private bool isMoving;
    
    public bool IsMoving => isMoving;
    
    public void Initialize(float speed)
    {
        moveSpeed = speed;
        currentSpeed = speed;
        agent = GetComponent<NavMeshAgent>();
        unit = GetComponent<Unit>();
        
        if (agent != null)
        {
            agent.speed = currentSpeed;
            agent.angularSpeed = 120f;
            agent.acceleration = 8f;
            agent.autoBraking = true;
        }
    }
    
    public void SetMoveSpeed(float speed)
    {
        currentSpeed = speed;
        if (agent != null)
        {
            agent.speed = currentSpeed;
        }

        if (agent == null)

    }
    
    public void Stop()

        isMoving = false;
```

#### 2.3.1 UnitCombat
```csharp
public class UnitCombat : MonoBehaviour
{
    private UnitStats stats;
    private Unit unit;
    private Unit target;
    private float attackTimer;
    private float damageMultiplier = 1f;
    private float attackSpeedMultiplier = 1f;
    private bool isAttacking;
    
    public Unit Target => target;
    public bool IsAttacking => isAttacking;
    
    public void Initialize(UnitStats stats)
    {
        this.stats = stats;
        unit = GetComponent<Unit>();
        attackTimer = 0f;
    }
    
    public void SetDamageMultiplier(float multiplier)
    {
        damageMultiplier = multiplier;
    }
    
    public void SetAttackSpeedMultiplier(float multiplier)
    {
        attackSpeedMultiplier = multiplier;
    }
    
    public void SetTarget(Unit newTarget)
    {
        target = newTarget;
    }
    
    public void ClearTarget()
    {
        target = null;
        isAttacking = false;
    }
    
    public bool CanAttack(Unit potentialTarget)
    {
        if (potentialTarget == null || potentialTarget.IsDead)
            return false;
        
        // 检查目标类型
        if (!IsValidTarget(potentialTarget))
            return false;
        
        // 检查距离
        float distance = Vector3.Distance(unit.transform.position, potentialTarget.transform.position);
        return distance <= stats.attackRange;
    }
    
    private bool IsValidTarget(Unit potentialTarget)
    {
        switch (stats.targetType)
        {
            case TargetType.Ground:
                return !potentialTarget.Stats.canFly;
            case TargetType.Air:
                return potentialTarget.Stats.canFly;
            case TargetType.Building:
                return potentialTarget.Stats.unitType == UnitType.Building;
            case TargetType.All:
                return true;
            default:
                return false;
        }
    }
    
    void Update()
    {
        if (target == null || target.IsDead || !CanAttack(target))
        {
            isAttacking = false;
            return;
        }
        
        attackTimer += Time.deltaTime;
        
        float attackInterval = stats.attackSpeed / attackSpeedMultiplier;
        
        if (attackTimer >= attackInterval)
        {
            Attack();
            attackTimer = 0f;
        }
    }
    
    private void Attack()
    {
        isAttacking = true;
        
        int damage = CalculateDamage();
        
        switch (stats.attackType)
        {
            case AttackType.SingleTarget:
                AttackSingleTarget(target, damage);
                break;
            case AttackType.Splash:
                AttackSplash(target.transform.position, damage);
                break;
            case AttackType.Line:
                AttackLine(target.transform.position, damage);
                break;
            case AttackType.Cone:
                AttackCone(target.transform.position, damage);
                break;
        }
        
        GameEvents.OnUnitAttacked?.Invoke(unit, target);
    }
    
    private int CalculateDamage()
    {
        int baseDamage = stats.attackDamage;
        
        // 应用伤害倍率
        baseDamage = Mathf.RoundToInt(baseDamage * damageMultiplier);
        
        // 检查冲锋伤害
        if (stats.isCharging && !hasCharged)
        {
            baseDamage = Mathf.RoundToInt(baseDamage * stats.chargeDamageMultiplier);
            hasCharged = true;
        }
        
        return baseDamage;
    }
    
    private bool hasCharged = false;
    
    private void AttackSingleTarget(Unit target, int damage)
    {
        target.TakeDamage(damage, stats.damageType);
    }
    
    private void AttackSplash(Vector3 center, int damage)
    {
        Collider[] colliders = Physics.OverlapSphere(center, stats.splashRadius);
        foreach (var collider in colliders)
        {
            Unit targetUnit = collider.GetComponent<Unit>();
            if (targetUnit != null && IsValidTarget(targetUnit))
            {
                float distance = Vector3.Distance(center, targetUnit.transform.position);
                float damageMultiplier = 1f - (distance / stats.splashRadius);
                int finalDamage = Mathf.RoundToInt(damage * damageMultiplier);
                targetUnit.TakeDamage(finalDamage, stats.damageType);
            }
        }
    }
    
    private void AttackLine(Vector3 direction, int damage)
    {
        Vector3 lineStart = unit.transform.position;
        Vector3 lineEnd = direction;
        
        RaycastHit[] hits = Physics.RaycastAll(lineStart, direction, stats.attackRange);
        foreach (var hit in hits)
        {
            Unit targetUnit = hit.collider.GetComponent<Unit>();
            if (targetUnit != null && IsValidTarget(targetUnit))
            {
                targetUnit.TakeDamage(damage, stats.damageType);
            }
        }
    }
    
    private void AttackCone(Vector3 direction, int damage)
    {
        Vector3 center = unit.transform.position;
        float coneAngle = 45f;
        
        Collider[] colliders = Physics.OverlapSphere(center, stats.attackRange);
        foreach (var collider in colliders)
        {
            Unit targetUnit = collider.GetComponent<Unit>();
            if (targetUnit != null && IsValidTarget(targetUnit))
            {
                Vector3 toTarget = (targetUnit.transform.position - center).normalized;
                float angle = Vector3.Angle(direction, toTarget);
                
                if (angle <= coneAngle / 2f)
                {
                    targetUnit.TakeDamage(damage, stats.damageType);
                }
            }
        }
    }
}
```
