```csharp
public class LODController : MonoBehaviour
{
    [System.Serializable]
    public class LODLevel
    {
        public float distance;
        public GameObject lodObject;
    }
    
    [SerializeField] private LODLevel[] lodLevels;
    private LODLevel currentLOD;
    
    void Update()
    {
        float distance = Vector3.Distance(Camera.main.transform.position, transform.position);
        LODLevel bestLOD = null;
        
        foreach (var lod in lodLevels)
        {
            if (distance <= lod.distance)
            {
                if (bestLOD == null || lod.distance < bestLOD.distance)
                {
                    bestLOD = lod;
                }
            }
        }
        
        if (bestLOD != currentLOD)
        {
            SwitchLOD(bestLOD);
        }
    }
    
    private void SwitchLOD(LODLevel newLOD)
    {
        if (currentLOD != null && currentLOD.lodObject != null)
        {
            currentLOD.lodObject.SetActive(false);
        }
        
        currentLOD = newLOD;
        
        if (currentLOD != null && currentLOD.lodObject != null)
        {
            currentLOD.lodObject.SetActive(true);
        }
    }
}
```

```csharp
public class TextureAtlasGenerator : MonoBehaviour

    public Texture2D GenerateAtlas(List<Texture2D> textures, int atlasSize = 4096)
    {
        Texture2D atlas = new Texture2D(atlasSize, atlasSize);
        atlas.wrapMode = TextureWrapMode.Clamp;
        atlas.filterMode = FilterMode.Bilinear;
        
        int currentX = 0;
        int currentY = 0;
        int maxHeight = 0;
        
        foreach (var texture in textures)
        {
            if (currentX + texture.width > atlasSize)
            {
                currentX = 0;
                currentY += maxHeight;
                maxHeight = 0;
            }
            
            atlas.SetPixels(currentX, currentY, texture.width, texture.height, texture.GetPixels());
            
            currentX += texture.width;
            maxHeight = Mathf.Max(maxHeight, texture.height);
        }
        
        atlas.Apply();
        return atlas;
    }
}
```

```csharp
    {
        yield return new WaitForSeconds(delay);
        
        particle.SetActive(false);

    }
}
```

```csharp
               (layer1 == 8 && layer2 == 0) ||
               (layer1 == 0 && layer2 == 9) || // 单位与建筑
```

#### 2.2.2 碰撞检测优化
```csharp
public class CollisionOptimizer : MonoBehaviour
{
    private List<Unit> allUnits = new List<Unit>();
    private float lastUpdateTime;
    private float updateInterval = 0.1f;
    
    void Update()
    {
        if (Time.time - lastUpdateTime < updateInterval)
            return;
        
        lastUpdateTime = Time.time;
        UpdateCollisions();
    }
    
    private void UpdateCollisions()
    {
        // 使用空间分区优化碰撞检测
        SpatialPartition partition = new SpatialPartition(10f);
        
        foreach (var unit in allUnits)
        {
            partition.Add(unit);
        }
        
        foreach (var unit in allUnits)
        {
            List<Unit> nearbyUnits = partition.Query(unit.transform.position, 5f);
            
            foreach (var nearbyUnit in nearbyUnits)
            {
                if (unit != nearbyUnit)
                {
                    CheckCollision(unit, nearbyUnit);
                }
            }
        }
    }
    
    private void CheckCollision(Unit unit1, Unit unit2)
    {
        float distance = Vector3.Distance(unit1.transform.position, unit2.transform.position);
        float minDistance = 1f;
        
        if (distance < minDistance)
        {
            // 处理碰撞
            Vector3 direction = (unit1.transform.position - unit2.transform.position).normalized;
            unit1.transform.position += direction * (minDistance - distance) * 0.5f;
            unit2.transform.position -= direction * (minDistance - distance) * 0.5f;
        }
    }
}

public class SpatialPartition
{
    private float cellSize;
    private Dictionary<Vector2Int, List<Unit>> grid = new Dictionary<Vector2Int, List<Unit>>();
    
    public SpatialPartition(float cellSize)
    {
        this.cellSize = cellSize;
    }
    
    public void Add(Unit unit)
    {
        Vector2Int cell = GetCell(unit.transform.position);
        
        if (!grid.ContainsKey(cell))
        {
            grid[cell] = new List<Unit>();
        }
        
        grid[cell].Add(unit);
    }
    
    public List<Unit> Query(Vector3 position, float radius)
    {
        List<Unit> result = new List<Unit>();
        int cellRadius = Mathf.CeilToInt(radius / cellSize);
        Vector2Int centerCell = GetCell(position);
        
        for (int x = -cellRadius; x <= cellRadius; x++)
        {
            for (int y = -cellRadius; y <= cellRadius; y++)
            {
                Vector2Int cell = new Vector2Int(centerCell.x + x, centerCell.y + y);
                
                if (grid.ContainsKey(cell))
                {
                    result.AddRange(grid[cell]);
                }
            }
        }
        
        return result;
    }
    
    private Vector2Int GetCell(Vector3 position)
    {
        return new Vector2Int(
            Mathf.FloorToInt(position.x / cellSize),
            Mathf.FloorToInt(position.z / cellSize)
        );
    }
}
```

```csharp
public class AsyncResourceLoader : MonoBehaviour
{
    public static AsyncResourceLoader Instance { get; private set; }
    
    private Dictionary<string, Object> loadedResources = new Dictionary<string, Object>();
    private Dictionary<string, List<Action<Object>>> loadingCallbacks = new Dictionary<string, List<Action<Object>>>();
    
    void Awake()
    {
        Instance = this;
    }
    
    public void LoadResourceAsync<T>(string path, Action<T> callback) where T : Object
    {
        if (loadedResources.ContainsKey(path))
        {
            callback?.Invoke(loadedResources[path] as T);
            return;
        }
        
        if (loadingCallbacks.ContainsKey(path))
        {
            loadingCallbacks[path].Add(obj => callback?.Invoke(obj as T));
            return;
        }
        
        loadingCallbacks[path] = new List<Action<Object>>();
        loadingCallbacks[path].Add(obj => callback?.Invoke(obj as T));
        
        StartCoroutine(LoadResourceCoroutine<T>(path));
    }
    
    private IEnumerator LoadResourceCoroutine<T>(string path) where T : Object
    {
        ResourceRequest request = Resources.LoadAsync<T>(path);
        
        yield return request;
        
        if (request.asset != null)
        {
            loadedResources[path] = request.asset;
            
            if (loadingCallbacks.ContainsKey(path))
            {
                foreach (var callback in loadingCallbacks[path])
                {
                    callback?.Invoke(request.asset);
                }
                
                loadingCallbacks.Remove(path);
            }
        }
        else
        {
            Debug.LogError($"Failed to load resource at path: {path}");
        }
    }
    
    public void UnloadResource(string path)
    {
        if (loadedResources.ContainsKey(path))
        {
            Resources.UnloadAsset(loadedResources[path]);
            loadedResources.Remove(path);
        }
    }
    
    public void UnloadUnusedResources()
    {
        Resources.UnloadUnusedAssets();
    }
}
```

```csharp
public class NetworkStateCompressor : MonoBehaviour
{
    public byte[] CompressState(NetworkState state)
    {
        using (MemoryStream ms = new MemoryStream())
        {
            using (BinaryWriter writer = new BinaryWriter(ms))
            {
                writer.Write(state.unitCount);
                
                foreach (var unit in state.units)
                {
                    // 使用16位整数压缩位置
                    writer.Write((short)(unit.position.x * 100));
                    writer.Write((short)(unit.position.y * 100));
                    writer.Write((short)(unit.position.z * 100));
                    
                    // 使用16位整数压缩旋转
                    writer.Write((short)(unit.rotation.eulerAngles.y * 100));
                    
                    // 使用字节压缩健康值
                    writer.Write((byte)(unit.healthPercent * 255));
                }
            }
            
            return ms.ToArray();
        }
    }
    
    public NetworkState DecompressState(byte[] data)
    {
        NetworkState state = new NetworkState();
        
        using (MemoryStream ms = new MemoryStream(data))
        {
            using (BinaryReader reader = new BinaryReader(ms))
            {
                state.unitCount = reader.ReadInt32();
                state.units = new UnitNetworkData[state.unitCount];
                
                for (int i = 0; i < state.unitCount; i++)
                {
                    state.units[i] = new UnitNetworkData
                    {
                        position = new Vector3(
                            reader.ReadInt16() / 100f,
                            reader.ReadInt16() / 100f,
                            reader.ReadInt16() / 100f
                        ),
                        rotation = Quaternion.Euler(0, reader.ReadInt16() / 100f, 0),
                        healthPercent = reader.ReadByte() / 255f
                    };
                }
            }
        }
        
        return state;
    }
}
```

```csharp
public class DeltaSync : MonoBehaviour
{
    private NetworkState lastState;
    private float syncInterval = 0.1f;
    private float lastSyncTime;
    
    void Update()
    {
        if (Time.time - lastSyncTime < syncInterval)
            return;
        
        lastSyncTime = Time.time;
        SendDeltaSync();
    }
    
    private void SendDeltaSync()
    {
        NetworkState currentState = GetCurrentState();
        NetworkDelta delta = CalculateDelta(lastState, currentState);
        
        if (delta.HasChanges())
        {
            SendDelta(delta);
            lastState = currentState;
        }
    }
    
    private NetworkDelta CalculateDelta(NetworkState oldState, NetworkState newState)
    {
        NetworkDelta delta = new NetworkDelta();
        delta.changedUnits = new List<UnitDelta>();
        
        for (int i = 0; i < newState.units.Length; i++)
        {
            UnitNetworkData oldUnit = oldState?.units[i];
            UnitNetworkData newUnit = newState.units[i];
            
            if (oldUnit == null || HasUnitChanged(oldUnit, newUnit))
            {
                delta.changedUnits.Add(new UnitDelta
                {
                    unitId = i,
                    position = newUnit.position,
                    rotation = newUnit.rotation,
                    healthPercent = newUnit.healthPercent
                });
            }
        }
        
        return delta;
    }
    
    private bool HasUnitChanged(UnitNetworkData oldUnit, UnitNetworkData newUnit)
    {
        float positionThreshold = 0.1f;
        float rotationThreshold = 1f;
        float healthThreshold = 0.01f;
        
        return Vector3.Distance(oldUnit.position, newUnit.position) > positionThreshold ||
               Quaternion.Angle(oldUnit.rotation, newUnit.rotation) > rotationThreshold ||
               Mathf.Abs(oldUnit.healthPercent - newUnit.healthPercent) > healthThreshold;
    }
    
    private void SendDelta(NetworkDelta delta)
    {
        // 发送差异数据到网络
    }
}
```

```csharp
public class PerformanceProfiler : MonoBehaviour
{
    private Dictionary<string, float> executionTimes = new Dictionary<string, float>();
    private Dictionary<string, int> executionCounts = new Dictionary<string, int>();
    
    public void StartProfiling(string label)
    {
        executionTimes[label] = Time.realtimeSinceStartup;
    }
    
    public void EndProfiling(string label)
    {
        if (executionTimes.ContainsKey(label))
        {
            float executionTime = Time.realtimeSinceStartup - executionTimes[label];
            
            if (!executionCounts.ContainsKey(label))
            {
                executionCounts[label] = 0;
            }
            
            executionCounts[label]++;
            
            Debug.Log($"{label}: {executionTime * 1000:F2}ms (Count: {executionCounts[label]})");
        }
    }
    
    public void PrintReport()
    {
        Debug.Log("=== Performance Report ===");
        
        foreach (var kvp in executionTimes)
        {
            Debug.Log($"{kvp.Key}: {kvp.Value * 1000:F2}ms");
        }
    }
}
```
